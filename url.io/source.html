Cloudflare Worker Script

Copy this code and paste it into your Cloudflare Worker script editor. This handles the redirection logic for the links you create in the Dashboard. 


// Cloudflare Worker Script

export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);
    const path = url.pathname;

    // 1. Remove leading slash for key lookup
    let key = path.replace(/^\//, '');
    
    // 2. Handle the specific use case: /alias//match...
    // If the user has double slashes, normalize or pass through.
    // We check if the key contains a slash to determine if there's extra path.
    let extraPath = '';
    let mainKey = key;

    // Check for path separator (single or double slashes)
    const firstSlashIndex = key.indexOf('/');
    
    if (firstSlashIndex !== -1) {
        mainKey = key.substring(0, firstSlashIndex);
        // Keep everything after the key (including the slash)
        extraPath = key.substring(firstSlashIndex); 
    }

    // 3. Fetch the mapping from KV (or hardcode for testing if env.LINKS isn't ready)
    // In a real setup, bind a KV namespace named 'LINKS' in wrangler.toml
    let targetUrl = null;
    let options = {};

    if (env.LINKS) {
      const storedData = await env.LINKS.get(mainKey, { type: 'json' });
      if (storedData) {
        targetUrl = storedData.url;
        // If the user checked "Append Path" (preservePath: true)
        if (storedData.preservePath && extraPath) {
            // Clean up double slashes in target if needed, or just append
            // If target ends with / and extra starts with //, clean it.
            // But per user request, they might want specific structure.
            // We simply join them:
            targetUrl = targetUrl.replace(/\/+$/, '') + extraPath;
        }
      }
    } else {
        // FALLBACK FOR TESTING WITHOUT KV:
        // This allows you to test the worker logic locally or before binding KV
        // by returning a hardcoded response for the example.
        if (mainKey === 'in-mc-fdlive-fc') {
            targetUrl = 'https://allinonereborn.online/fcdata/playlists' + extraPath;
        }
    }

    // 4. Perform Redirect
    if (targetUrl) {
      return Response.redirect(targetUrl, 301);
    }

    // 5. Default Response (Not Found)
    return new Response('404 - Link not found', { status: 404 });
  },
};

Note: For production, create a KV Namespace in Cloudflare, bind it as `LINKS` in your `wrangler.toml`, and update your Dashboard to write to that KV namespace (this HTML tool currently uses LocalStorage for UI management).
